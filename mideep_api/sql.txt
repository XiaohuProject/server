# profile
$ create table profile
(
     uid int not null auto_increment,
     name varchar(100) not null,
     cid varchar(100) not null,
     wid varchar(100) not null,
     enter_time datetime not null,
     gender varchar(8) not null,
     primary key(uid)
)default charset=utf8;
